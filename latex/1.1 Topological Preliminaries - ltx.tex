\begin{definition}\label{def:topology}
A \underline{topology} $\mathcal{T}$ in a set $X$ is a collection of subsets of $X$ (called open sets) satisfying the following axioms:
\begin{enumerate}[(1)]
\item the union of any number of open sets is open;
\item the intersection of any finite number of open sets is open;
\item $X, \emptyset$ are open.
\end{enumerate}
A \underline{topological space} is a set $X$ with a topology $\mathcal{T}$ in $X$.

$\mathcal{T} = \{ \emptyset, X \}$ is the \underline{trivial topology}.

$\mathcal{T} = \mathcal{P}(X)$ is the \underline{discrete topology}.
\end{definition}

\begin{definition}\label{def:induced_topology}
For $A \subseteq X$, where $\mathcal{T}$ is a topology in $X$, then $\mathcal{T}_A$ is an \underline{induced topology} in $A$.
\end{definition}

% --- Page 2 ---
\[
\mathcal{T}_A=\{U\cap A\mid U\in\mathcal{T}\}.
\]

\begin{definition}\label{def:closed_set}
In a topological space, a closed set is a complement of an open set.
\end{definition}

\begin{definition}\label{def:closure}
For $E\subseteq X$, $\overline E$ is said to be the closure of $E$, where
\[
\overline E=\bigcap\{V\mid V\text{ is closed and }V\supseteq E\}.
\]
\end{definition}

\begin{definition}\label{def:neighborhood}
A neighborhood of $x\in X$ is any open set containing $x$.
\end{definition}

\begin{definition}\label{def:mapping}
A mapping $f$ from $X_1$ to $X_2$ ($X_1\xrightarrow{f}X_2$) is an assignment to each $x\in X_1$ of an element $f(x)\in X_2$ where $X_1,X_2$ are topological spaces. (Map and mapping are not the same; a mapping is a function between two topological spaces.)
\end{definition}

% --- Page 3 ---
\begin{definition}\label{def:continuous}
The mapping $f$ is continuous if and only if for every open set $O_2$ in $X_2$, the set $f^{-1}(O_2)$ is an open set in $X_1$. (Note $f^{-1}$ need not be a mapping.)
\end{definition}

\begin{definition}\label{def:continuous_at_point}
For $x\in X_1$, we say that $f$ is continuous at $x$ if for every neighborhood $U$ of $f(x)$, there exists a neighborhood $V$ of $x$ such that $f(V)\subset U$. ($f$ is continuous if and only if it is continuous at every point of $X_1$.)
\end{definition}

We can show the equivalence of both definitions.

\begin{lemma}\label{lem:composition_continuous}
The composition of continuous maps is continuous.
\end{lemma}

\begin{proof}
If $f\colon X\to Y$, $g\colon Y\to Z$ are continuous.

Goal 1: $(g\circ f)^{-1}(V)$ is open in $X$ where $V$ is any open set in $Z$.

Goal 2:
\[
(g\circ f)^{-1}(V)=f^{-1}\circ g^{-1}(V).
\]
\end{proof}

\item If $X$ is a topological space and $Y$ is a set with the trivial topology, then $f\colon X\rightarrow Y$ is continuous.

\begin{definition}\label{def:projections}
If
\[
p_1:X_1\times X_2\rightarrow X_1
\quad\text{and}\quad
p_2:X_1\times X_2\rightarrow X_2
\]
are such that
\[
p_1(x_1\times x_2)=x_1,
\qquad
p_2(x_1\times x_2)=x_2,
\]
respectively, then $p_1,p_2$ are projections of $X_1\times X_2$.
\end{definition}

% --- Page 5 ---
\begin{lemma}\label{lem:projections_continuous_open}
Projections are continuous, open mappings.
\end{lemma}

\begin{proof}
It suffices to show this for $p_1$; for $p_2$ the argument is the same.

For the open map property, observe every open set in $X$ can be written as $U\times V$ where $U$ is an open set in $X_1$, and $V$ is an open set in $X_2$.

Now,
\[
p_1(U\times V)=U.
\]

For continuity, observe
\[
p_1^{-1}(U)=U\times X_2
\]
where $U\subset X_1$ is open.
\end{proof}

\begin{definition}\label{def:covering}
A family of subsets $\{V_\alpha\}_{\alpha\in I}$ is a \underline{covering} if
\[
\bigcup_{\alpha\in I}V_\alpha=X.
\]
If each $V_\alpha$ is an open subset of $X$, $\{V_\alpha\}_{\alpha\in I}$ is an \underline{open covering}, and if $|I|<+\infty$ then $\{V_\alpha\}_{\alpha\in I}$ is a finite covering.
\end{definition}

% --- Page 6 ---
\begin{definition}\label{def:compact}
A topological space $X$ is \textbf{compact} if every open covering of $X$ contains a finite covering. A subset $A$ is said to be compact if $A$ is compact in the induced topology.
\end{definition}

\begin{remark}\label{rem:compactness_examples}
\begin{enumerate}
\item $\mathbb R$ is not compact.
\item If $A\subseteq X$, and $|A|<+\infty$, then $A$ is compact.
\item Let $a,b\in\mathbb R$, $a\le b$, then $[a,b]$ is compact in $\mathbb R$.
\end{enumerate}
\end{remark}

\begin{proof}
Given an open covering of $[a,b]$, say $\mathcal A$.

STEP 1: For $x\in[a,b]$ there exists $y\in[a,b]$ and $y>x$ such that $[a,y]$ is covered by at most two elements of $\mathcal A$. (Do the case where an immediate successor exists and the case where an immediate successor does not exist.)

STEP 2:
\[
C=\{y\in[a,b]\mid [a,y]\text{ is covered by finitely many elements of }\mathcal A\}.
\]

Observe $C\neq\emptyset$ by applying STEP 1 for $x=a$, and apply the least upper bound property where
\[
c=\sup C.
\]

STEP 3: Show $c\in C$.

STEP 4: $b\in C$.
\end{proof}

\begin{remark}\label{rem:finite_intersection_property}
\textbf{Finite intersection property}: In a topological space $X$, let $\{F_\lambda\}$ be a collection of closed sets. If the intersection of any finite subcollection is non-empty, then
\[
\bigcap_\lambda F_\lambda\neq\emptyset.
\]
A space has the finite intersection property if and only if the space is compact.
\end{remark}

\begin{theorem}\label{thm:closed_subset_compact}
A closed subset of a compact set is compact.
\end{theorem}

\begin{proof}
Let $X$ be compact, and $Y$ be a closed subset of $X$.

For any open covering of $Y$, say $\mathcal A$, we construct
\[
\mathcal A\cup\{X\setminus Y\}
\]
as an open covering of $X$, as $X\setminus Y$ is open.

As $X$ is compact, let $\mathcal A'$ be a finite subcover of $X$.

Now, the finite subcover of $Y$ we get from $\mathcal A'$ will not contain $X\setminus Y$, hence it is a finite subcover of $\mathcal A$.
\end{proof}

\begin{definition}\label{def:separated_open_sets}
Two sets $M_1$ and $M_2$ are separated by open sets if there exist open sets $O_1$ and $O_2$ containing $M_1$ and $M_2$, respectively, such that
\[
O_1\cap O_2=\emptyset.
\]
\end{definition}

\begin{definition}\label{def:separated_real_function}
Two sets $M_1$ and $M_2$ are separated by a real function if there exists a continuous real function
\[
f\colon X\rightarrow[0,1]
\]
such that $f(x)=0$ for $x\in M_1$ and $f(x)=1$ for $x\in M_2$.
\end{definition}

\begin{definition}\label{def:separation_axioms}
\begin{itemize}
\item $T_1$: For any disjoint points, there exists an open set containing one and not the other. Equivalently, all singletons are closed.
\item $T_2$ (Hausdorff): Two distinct points can be separated by open sets.
\item $T_3$ (Regular): For a closed set $F$ and $x\notin F$, $\{x\}$ and $F$ can be separated by open sets.
\item $T_4$ (Normal): Disjoint closed sets can be separated
% --- Page 12 ---
by open sets.
\end{itemize}
\end{definition}

\begin{remark}\label{rem:separation_implications}
$T_4\Rightarrow T_3\Rightarrow T_2\Rightarrow T_1$.
\end{remark}
