\begin{itemize}
    \item A \underline{topology} $T$ in a set $X$ is a collection of subsets of $X$ (called open sets) satisfying the following axioms:
    \begin{enumerate}
        \item the union of any number of open sets is open
        \item the intersection of any finite number of open sets is open
        \item $X$, $\emptyset$ are open.
    \end{enumerate}

    A \underline{topological space} is a set $X$ with a topology $T$ in $X$.

    \begin{remark}
        $T = \{\emptyset, X\}$ is \underline{trivial topology}

        $T = \mathcal{P}(X)$ is \underline{discrete topology}
    \end{remark}

    \item For $A \subseteq X$, where $T$ is a topology in $X$. Then $T_A$ is an \underline{induced topology} in $A$
\end{itemize}

% --- Page 2 ---
\[ T_A = \bigcap \{ U \cap A \mid U \in \mathcal{T}_X \} \]

\begin{itemize}
    \item In a topological spaces, a \textbf{closed set} is a complement of an open set.
    \item For \( E \subseteq X \), \( \overline{E} \) is said to be \textbf{closure} of \( E \), where \( \overline{E} = \bigcap \{ V \mid V \text{ is closed and } V \supseteq E \} \).
    \item A \textbf{neighbourhood} of \( x \in X \), is any open set containing \( x \).
    \item A \textbf{mapping} \( f \) from \( X_1 \) to \( X_2 \) (\( X_1 \xrightarrow{f} X_2 \)) is an assignment of each \( x \in X_1 \) of an element \( f(x) \in X_2 \) where \( X_1, X_2 \) are topological spaces. (map and mapping are not the same, mapping is a function between two topological spaces).
\end{itemize}

\begin{itemize}
    \item The mapping $f$ is \underline{continuous} if and only if for every open set $O_2$ in $X_2$, the set $f^{-1}(O_2)$ is an open set in $X_1$. (note $f^{-1}$ need not be a mapping)

    \item For $x \in X_1$, we say that $f$ is continuous at $x$, if to every neighbourhood $U$ of $f(x)$, there exists a nbhd $V$ of $x$, such that $f(V) \subset U$. (f is continuous at every point of $X_1$)

    We can show the equivalence of both definitions
\end{itemize}

\begin{lemma}
    Composition of continuous maps are continuous.
\end{lemma}

\begin{proof}
    If $f: X \rightarrow Y$, $g: Y \rightarrow Z$ are continuous.

    Goal 1: $(g \circ f)^{-1}(V)$ is open in $X$ where $V$ is any open set in $Z$.

    Goal 2: $(g \circ f)^{-1}(V) = f^{-1} \circ g^{-1}(V)$
\end{proof}

\begin{theorem}
A closed subset of a compact set is compact
\end{theorem}

\begin{proof}
Let $X$ be compact, and $Y$ be a closed subset of $X$.

For any open covering of $Y$, say $\mathcal{A}$, we construct $\mathcal{A} \cup \{X \setminus Y\}$ as an open covering of $X$ as $X \setminus Y$ is open. As $X$ is compact, let $\hat{\mathcal{A}}$ be a finite subcover of $X$. Now, the finite subcover of $Y$ we get from $\hat{\mathcal{A}}$ will not contain $X \setminus Y$, hence finite subcover of $\mathcal{A}$.
\end{proof}

\begin{definition}
Let $X$ be a topological space. A subset $A \subseteq X$ is called
\begin{itemize}
    \item \textit{open} if $A$ is an open set in $X$,
    \item \textit{closed} if its complement $X \setminus A$ is open,
    \item \textit{compact} if every open cover of $A$ has a finite subcover,
    \item \textit{connected} if there do not exist open sets $U, V \subseteq X$ such that $A \subseteq U \cup V$, $A \cap U \neq \emptyset$, $A \cap V \neq \emptyset$, and $U \cap V = \emptyset$.
\end{itemize}
\end{definition}

\begin{theorem}
Let $X$ be a topological space and $A \subseteq X$. Then the following are equivalent:
\begin{enumerate}
    \item $A$ is compact.
    \item Every family of closed subsets of $A$ with the finite intersection property has non-empty intersection.
    \item Every ultrafilter on $A$ has a non-empty intersection.
\end{enumerate}
\end{theorem}

\begin{lemma}
Let $X$ be a topological space and $A \subseteq X$ be compact. If $f: X \to Y$ is a continuous function and $Y$ is a Hausdorff space, then $f(A)$ is compact in $Y$.
\end{lemma}

\begin{proof}
Let $\{V_\alpha\}$ be an open cover of $f(A)$. Then $\{f^{-1}(V_\alpha)\}$ is an open cover of $A$. Since $A$ is compact, there exists a finite subcover $\{f^{-1}(V_{\alpha_1}), \ldots, f^{-1}(V_{\alpha_n})\}$ of $A$. Then $\{V_{\alpha_1}, \ldots, V_{\alpha_n}\}$ is a finite subcover of $f(A)$. Hence, $f(A)$ is compact.
\end{proof}

\begin{remark}
The above lemma is often referred to as the \textit{closed map property} for compact sets in Hausdorff spaces.
\end{remark}

\begin{definition}
A topological space $X$ is called \textit{Hausdorff} if for any two distinct points $x, y \in X$, there exist disjoint open sets $U, V \subseteq X$ such that $x \in U$ and $y \in V$.
\end{definition}

\begin{theorem}
Let $X$ be a compact Hausdorff space. Then every closed subset of $X$ is compact.
\end{theorem}

\begin{proof}
Let $A \subseteq X$ be closed. Since $X$ is compact, $A$ is compact if it satisfies the definition of compactness. Let $\{U_\alpha\}$ be an open cover of $A$. Then $\{U_\alpha\} \cup \{X \setminus A\}$ is an open cover of $X$. Since $X$ is compact, there exists a finite subcover $\{U_{\alpha_1}, \ldots, U_{\alpha_n}, X \setminus A\}$. Removing $X \setminus A$ from this subcover leaves a finite subcover of $A$. Hence, $A$ is compact.
\end{proof}

\begin{remark}
In a Hausdorff space, compact sets are closed. This is a useful property when dealing with compactness in Hausdorff spaces.
\end{remark}
