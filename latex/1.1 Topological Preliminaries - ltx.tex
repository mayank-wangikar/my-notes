\begin{itemize}
    \item A \underline{topology} $T$ in a set $X$ is a collection of subsets of $X$ (called open sets) satisfying the following axioms:
    \begin{enumerate}
        \item the union of any number of open sets is open
        \item the intersection of any finite number of open sets is open
        \item $X$, $\emptyset$ are open.
    \end{enumerate}

    A \underline{topological space} is a set $X$ with a topology $T$ in $X$.

    \begin{itemize}
        \item $T = \{ \emptyset, X \}$ is \underline{trivial topology}
        \item $T = \mathcal{P}(X)$ is \underline{discrete topology}
    \end{itemize}

    \item For $A \subseteq X$, where $T$ is a topology in $X$. Then $T_A$ is an \underline{induced topology} in $A$
\end{itemize}

\begin{itemize}
    \item In a topological spaces, a \textbf{closed set} is a complement of an open set.

    \item For $E \subseteq X$, $\overline{E}$ is said to be \textbf{closure} of $E$, where $\overline{E} = \bigcap \{ V \mid V \text{ is closed and } V \supseteq E \}$, $S$ is the set of all closed sets containing $E$.

    \item A \textbf{neighbourhood} of $x \in X$, is any open set containing $x$.

    \item A \textbf{mapping} $f$ from $X_1$ to $X_2$ ($X_1 \xrightarrow{f} X_2$) is an assignment of each $x \in X_1$ of an element $f(x) \in X_2$ where $X_1$, $X_2$ are topological spaces. (map and mapping are not the same, mapping is a function between two topological spaces).
\end{itemize}

% --- Page 3 ---
\begin{itemize}
    \item The mapping $f$ is \underline{continuous} if and only if for every open set $O_2$ in $X_2$, the set $f^{-1}(O_2)$ is an open set in $X_1$. (note $f^{-1}$ need not be a mapping)

    \item For $x \in X_1$, we say that $f$ is continuous at $x$, if to every neighbourhood $(U)$ of $f(x)$, there exists a nbd $V$ of $x$, such that $f(V) \subset U$. ($f$ is continuous at every point of $X_1$.)
\end{itemize}

We can show the equivalence of both definitions

\begin{lemma}
    Composition of continuous maps are continuous.
\end{lemma}

\begin{proof}
    If $f: X \rightarrow Y$, $g: Y \rightarrow Z$ are continuous.

    \begin{itemize}
        \item Goal 1: $(g \circ f)^{-1}(V)$ is open in $X$ where $V$ is any open set in $Z$.
        \item Goal 2: $(g \circ f)^{-1}(V) = f^{-1} \circ g^{-1}(V)$
    \end{itemize}
\end{proof}

\begin{theorem}
A closed subset of a compact set is compact
\end{theorem}

\begin{proof}
Let $X$ be compact, and $Y$ be a closed subset of $X$.

For any open covering of $Y$, say $\mathcal{A}$, we construct $\mathcal{A} \cup \{X \setminus Y\}$ as an open covering of $X$ as $X \setminus Y$ is open. As $X$ is compact, let $\hat{\mathcal{A}}$ be a finite subcover of $X$. Now, the finite subcover of $Y$ we get from $\hat{\mathcal{A}}$ will not contain $X \setminus Y$, hence finite subcover of $\mathcal{A}$.
\end{proof}

\begin{definition}
Two sets \( M_1 \) and \( M_2 \) are \underline{separated by open sets} if there exist open sets \( O_1 \) and \( O_2 \) containing \( M_1 \) and \( M_2 \) such that \( O_1 \cap O_2 = \emptyset \).
\end{definition}

\begin{definition}
Two sets \( M_1 \) and \( M_2 \) are \underline{separated by a real function} if there exists a continuous real function \( f: X \rightarrow [0,1] \) such that \( f(x) = 0 \) for \( x \in M_1 \) and \( f(x) = 1 \) for \( x \in M_2 \).
\end{definition}

\begin{definition} ($T_1$)
For any disjoint points, there exists an open set containing one and not the other. Equivalently, all singletons are closed.
\end{definition}

\begin{definition} ($T_2$ (Hausdorff))
Two distinct points can be separated by open sets.
\end{definition}

\begin{definition} ($T_3$ (Regular))
For closed set \( F \) and \( x \notin F \), can be separated by open sets.
\end{definition}

\begin{definition} ($T_4$ (Normal))
Disjoint closed sets can be separated by open sets.
\end{definition}

\begin{definition}
Let $X$ be a topological space. A subset $A \subseteq X$ is called \textit{dense} in $X$ if $\overline{A} = X$, where $\overline{A}$ denotes the closure of $A$.
\end{definition}

\begin{theorem}
Let $X$ be a topological space. The following statements are equivalent for a subset $A \subseteq X$:
\begin{enumerate}
    \item $A$ is dense in $X$.
    \item For every non-empty open set $U \subseteq X$, $U \cap A \neq \emptyset$.
    \item The only dense subset of $X$ containing $A$ is $X$ itself.
\end{enumerate}
\end{theorem}

\begin{lemma}
Let $X$ be a metric space and $A \subseteq X$ be a subset. If $A$ is dense in $X$, then for every $x \in X$ and every $\epsilon > 0$, there exists $a \in A$ such that $d(x, a) < \epsilon$.
\end{lemma}

\begin{proof}
Let $x \in X$ and $\epsilon > 0$. Since $A$ is dense in $X$, the closure of $A$ is $X$. The open ball $B(x, \epsilon)$ is an open set in $X$. By the definition of density, $B(x, \epsilon) \cap A \neq \emptyset$. Therefore, there exists $a \in A$ such that $a \in B(x, \epsilon)$, which implies $d(x, a) < \epsilon$.
\end{proof}

\begin{remark}
The concept of density is crucial in analysis and topology. For instance, the set of rational numbers $\mathbb{Q}$ is dense in the real numbers $\mathbb{R}$ with the standard topology.
\end{remark}

Let $X$ be a topological space and $A \subseteq X$. Suppose $A$ is dense in $X$. Then for any continuous function $f: X \to Y$ where $Y$ is another topological space, the restriction of $f$ to $A$ determines $f$ on all of $X$ if $A$ is also dense in the domain of $f$ in $Y$ (this is not always true, but it holds under certain conditions).

Consider a sequence $\{x_n\}$ in $X$ such that $x_n \to x$ in $X$. If $A$ is dense in $X$, then for any $x \in X$, there exists a sequence $\{a_n\}$ in $A$ such that $a_n \to x$.
