\begin{definition}
A \textit{topology} $T$ in a set $X$ is a collection of subsets of $X$ (called open sets) satisfying the following axioms:
\begin{enumerate}
\item the union of any number of open sets is open
\item the intersection of any finite number of open sets is open.
\item $X$, $\emptyset$ are open.
\end{enumerate}
\end{definition}

A \textit{topological space} is a set $X$ with a topology $T$ in $X$.

\begin{remark}
$T = \{\emptyset, X\}$ is \textit{trivial topology}

$T = \mathcal{P}(X)$ is \textit{discrete topology}
\end{remark}

\begin{remark}
For $A \subseteq X$, where $T$ is a topology in $X$, then $T_A$ is an \textit{induced topology} in $A$.
\end{remark}

* A neighbourhood of \( x \in X \), is any open set containing \( x \).

* A mapping \( f \) from \( X_1 \) to \( X_2 \) (\( X_1 \xrightarrow{f} X_2 \)) is an assignment of each \( x \in X_1 \) of an element \( f(x) \in X_2 \) where \( X_1, X_2 \) are topological spaces. (map and mapping are not the same, mapping is a function between two topological spaces).

The mapping $f$ is continuous if and only if for every open set $O_2$ in $X_2$, the set $f^{-1}(O_2)$ is an open set in $X_1$. (note $f^{-1}$ need not be a mapping)

For $x \in X_1$, we say that $f$ is continuous at $x$, if to every neighbourhood $U$ of $f(x)$, there exists a nbd $V$ of $x$, such that $f(V) \subset U$. ($f$ is continuous at every point of $X_1$)

We can show the equivalence of both definitions

\begin{lemma}
Composition of continuous maps are continuous.
\end{lemma}

\begin{proof}
If $f: X \rightarrow Y$, $g: Y \rightarrow Z$ are continuous.

Goal 1: $(g \circ f)^{-1}(V)$ is open in $X$ where $V$ is any open set in $Z$.

Goal 2: $(g \circ f)^{-1}(V) = f^{-1} \circ g^{-1}(V)$
\end{proof}

\begin{theorem}
A closed subset of a compact set is compact
\end{theorem}

\begin{proof}
Let $X$ be compact, and $Y$ be a closed subset of $X$.

For any open covering of $Y$, say $\mathcal{A}$, we construct $\mathcal{A} \cup \{X \setminus Y\}$ as an open covering of $X$ as $X \setminus Y$ is open. As $X$ is compact, let $\hat{\mathcal{A}}$ be a finite subcover of $X$. Now, the finite subcover of $Y$ we get from $\hat{\mathcal{A}}$ will not contain $X \setminus Y$, hence finite subcover of $\mathcal{A}$.
\end{proof}

\begin{itemize}

    \item Two sets \( M_1 \) and \( M_2 \) are \textit{separated by open sets} if there exist open sets \( O_1 \) and \( O_2 \) containing \( M_1 \) and \( M_2 \) such that \( O_1 \cap O_2 = \emptyset \).

    \item Two sets \( M_1 \) and \( M_2 \) are \textit{separated by a real function} if there exists a continuous real function \( f: X \rightarrow [0,1] \) such that \( f(x) = 0 \) for \( x \in M_1 \) and \( f(x) = 1 \) for \( x \in M_2 \).

    \item \( T_1 \): For any disjoint points, there exists an open set containing one and not the other. Equivalently, all singletons are closed.

    \item \( T_2 \) (Hausdorff): Two distinct points can be separated by open sets.

    \item \( T_3 \) (Regular): For closed set \( F \) and \( x \notin F \), can be separated by open sets.

    \item \( T_4 \) (Normal): Disjoint closed sets can be separated
\end{itemize}

\begin{definition}
Let $X$ be a topological space. A subset $A \subseteq X$ is called a \textbf{closed set} if its complement $X \setminus A$ is an open set.
\end{definition}

\begin{theorem}
Let $f: X \to Y$ be a continuous function between topological spaces. If $A \subseteq X$ is a closed set, then $f(A)$ is a closed set in $Y$.
\end{theorem}

\begin{proof}
Since $A$ is closed in $X$, its complement $X \setminus A$ is open in $X$. By the continuity of $f$, the preimage of any open set in $Y$ is open in $X$. Thus, $f^{-1}(Y \setminus f(A)) = X \setminus A$ is open in $X$. This implies that $Y \setminus f(A)$ is open in $Y$, and hence $f(A)$ is closed in $Y$.
\end{proof}

\begin{lemma}
Let $X$ be a topological space and $\{A_i\}_{i \in I}$ be a collection of closed sets in $X$. Then the intersection $\bigcap_{i \in I} A_i$ is closed in $X$.
\end{lemma}

\begin{example}
Consider the real line $\mathbb{R}$ with the standard topology. The closed interval $[a, b]$ is a closed set in $\mathbb{R}$ since its complement $\mathbb{R} \setminus [a, b]$ is the union of two open sets $(-\infty, a)$ and $(b, \infty)$.
\end{example}

\begin{remark}
In a metric space, a set is closed if and only if it contains all its limit points.
\end{remark}

\begin{definition}
A subset $A$ of a topological space $X$ is called \textbf{compact} if every open cover of $A$ has a finite subcover.
\end{definition}

\begin{theorem}
Let $f: X \to Y$ be a continuous function between topological spaces. If $K \subseteq X$ is a compact set, then $f(K)$ is a compact set in $Y$.
\end{theorem}

\begin{proof}
Let $\{V_j\}_{j \in J}$ be an open cover of $f(K)$. Then $\{f^{-1}(V_j)\}_{j \in J}$ is an open cover of $K$. Since $K$ is compact, there exists a finite subcover $\{f^{-1}(V_{j_1}), \ldots, f^{-1}(V_{j_n})\}$ of $K$. It follows that $\{V_{j_1}, \ldots, V_{j_n}\}$ is a finite subcover of $f(K)$.
\end{proof}

\begin{lemma}
In a metric space, a set is compact if and only if it is complete and totally bounded.
\end{lemma}

\begin{example}
The closed interval $[0, 1]$ is a compact subset of $\mathbb{R}$ with the standard topology.
\end{example}

\begin{definition}
A function $f: X \to Y$ between topological spaces is called \textbf{homeomorphism} if $f$ is bijective, continuous, and its inverse $f^{-1}: Y \to X$ is also continuous.
\end{definition}

\begin{remark}
If there exists a homeomorphism between two topological spaces, then they are said to be \textbf{homeomorphic}.
\end{remark}
