Let $S^1$ be the unit circle in $\mathbb{R}^2$ with the standard coordinate system.

\begin{definition}\label{def:stereographic_projection}
Let $P_N$ denote a stereographic projection from $N = (0, 1)$.
The map is given by
\[
P_N \colon S^1 \setminus \{N\} \to \mathbb{R},
\]
\[
(x, y) \mapsto \frac{x}{1 - y}.
\]
\end{definition}

\begin{proof}[Derivation of the formula]
By the two-point formula, consider the line passing through $N = (0, 1)$ and a point $(x, y) \in S^1$. The $x$-intercept $(x_0, 0)$ of this line satisfies
\[
\frac{y - 1}{x - 0} = \frac{1 - 0}{0 - x_0}.
\]
Since $(x, y) \in S^1$, we have $x^2 + y^2 = 1$. Solving for $x_0$ gives
\[
x_0 = \frac{x}{1 - y}.
\]
\end{proof}

\begin{proposition}\label{prop:bijective_projection}
Check that $P_N$ is bijective.
\end{proposition}

\begin{proof}
Consider the kernel of $P_N$:
\[
\ker P_N = \left\{ \frac{x}{1 - y} = 0 \right\}.
\]
This implies $x = 0$ and $y \neq 1$. Since $(x, y) \in S^1$, we have $x^2 + y^2 = 1$. If $x = 0$, then $y = \pm 1$. However, $y = 1$ is excluded (as $N$ is the projection point), so the only possible point is $(0, -1)$. But $(0, -1)$ maps to $0$ under $P_N$, and no other point maps to $0$ except $(0, -1)$. Thus, $P_N$ is injective.

For surjectivity, given any $z \in \mathbb{R}$, we can find $(x, y) \in S^1$ such that $P_N(x, y) = z$. Solving for $y$ in terms of $x$ and $z$ yields a unique point on $S^1$ for each $z \in \mathbb{R}$. Hence, $P_N$ is bijective.
\end{proof}

\begin{definition}\label{def:chart}
Given a space $X$, $n \in \mathbb{N}$.
A \emph{chart} for $X$ is a pair $(U, \varphi)$ where $U$ is an open subset of $X$ and $\varphi \colon U \to A$ is a homeomorphism, where $A$ is an open subset of $\mathbb{R}^n$.
\end{definition}

\begin{definition}\label{def:compatible_chart}
For two charts $(U, \varphi)$ and $(V, \psi)$ are \emph{compatible} if $\varphi \circ \psi^{-1}$ and $\psi \circ \varphi^{-1}$ are differentiable (and invertible), and $U \cap V \neq \emptyset$.
\end{definition}

\begin{definition}\label{def:atlas}
An \emph{atlas} is a collection of compatible charts $\{(U_i, \varphi_i)\}_{i \in I}$ such that the union of $U_i$'s contains every point of $X$.
\end{definition}

\begin{remark}\label{rem:smooth_structure_intro}
By smooth structure we mean a way of doing calculus.
\end{remark}

\begin{definition}\label{def:smooth_structure}
A \emph{smooth structure} on $X$ is a maximal atlas (of compatible charts).
\end{definition}

Observe that $f$ is continuous on $[-1, 1]$ and differentiable on $(-1, 1)$.

\begin{proof}
By the Mean Value Theorem (MVT), consider $(x, x_0) \subseteq (-1, 1)$ such that $\exists c \in (x, x_0)$ with
\[
|f(x) - f(x_0)| = |f'(c)| \cdot |x - x_0|.
\]

We compute the derivative of $f$:
\[
f'(x) = \frac{-x}{(1 - x^2)^{1/2}}.
\]

Thus,
\[
|f'(c)| = \frac{|c|}{(1 - c^2)^{1/2}}.
\]

Since $c \in (x, x_0)$ and $|x - x_0| < \delta$, we have $|c - x_0| < \delta \implies |c| < (|x_0| + \delta)$.

Define
\[
M(x_0, \delta) = \frac{|x_0| + \delta}{(1 - (|x_0| + \delta)^2)^{1/2}} = M.
\]

Therefore,
\[
|f(x) - f(x_0)| < M \delta.
\]
\end{proof}

\begin{definition}

\[
\nabla g = \left( \frac{\partial g}{\partial x_1}, \ldots, \frac{\partial g}{\partial x_n} \right).

\begin{definition}
Let $h: \mathbb{R}^n \to \mathbb{R}^m$ be a differentiable function. The derivative of $h$ is given by the Jacobian matrix
\[
Dh = \begin{pmatrix}
\frac{\partial h_1}{\partial x_1} & \cdots & \frac{\partial h_1}{\partial x_n} \\
\vdots & \ddots & \vdots \\
\frac{\partial h_m}{\partial x_1} & \cdots & \frac{\partial h_m}{\partial x_n}
\end{pmatrix}_{m \times n}
=
\begin{pmatrix}
\frac{\partial h_i}{\partial x_j}
\end{pmatrix}_{m \times n}.
\]
\end{definition}

\begin{exercise}\label{ex:det_orthogonal}
Show that if $A \in O_n(\mathbb{R})$ and $\det(A) \neq 1$, then $\det(A) = -1$.
\end{exercise}

\begin{proof}
Since $A \in O_n(\mathbb{R})$, we have $A^T A = I$. Taking determinants on both sides, we get:
\[
\det(A^T A) = \det(I).
\]
Using the property of determinants, $\det(A^T) = \det(A)$ and $\det(I) = 1$, we have:
\[
\det(A^T) \det(A) = 1 \implies (\det(A))^2 = 1.
\]
Thus, $\det(A) = \pm 1$. Given that $\det(A) \neq 1$, it follows that $\det(A) = -1$.
\end{proof}

The wedge product $\wedge$ satisfies the following properties:

\begin{enumerate}
    \item
    \[
    w^k \wedge w^\ell = (-1)^{k\ell} w^\ell \wedge w^k
    \]

    \begin{proof}
    We express $w^k \wedge w^\ell$ as a summation over permutations $\sigma \in S_{k+\ell}$:
    \[
    w^k \wedge w^\ell = \sum_{\sigma \in S_{k+\ell}} \operatorname{sgn}(\sigma) \, w^k(V_{\sigma(1)} \cdots V_{\sigma(k)}) \, w^\ell(V_{\sigma(k+1)} \cdots V_{\sigma(k+\ell)}).
    \]
    Swapping the indices $k$ and $\ell$ introduces a sign factor $(-1)^{k\ell}$ due to the permutation of the combined set of indices. Thus,
    \[
    w^k \wedge w^\ell = (-1)^{k\ell} w^\ell \wedge w^k.
    \]
    \end{proof}

    \item
    \[
    (w^k \wedge w^\ell) \wedge w^m = w^k \wedge (w^\ell \wedge w^m).
    \]

    \begin{proof}
    By the associativity of the wedge product, we have
    \[
    (w^k \wedge w^\ell) \wedge w^m = \sum_{\sigma \in S_{k+\ell}} \operatorname{sgn}(\sigma) \, w^k(V_{\sigma(1)} \cdots V_{\sigma(k)}) \, w^\ell(V_{\sigma(k+1)} \cdots V_{\sigma(k+\ell)}) \wedge w^m.
    \]
    Expanding the wedge product further, we see that the associativity follows from the permutation properties of the wedge product itself, ensuring that the order of operations does not affect the final result.
    \end{proof}

    \item
    \[
    (\lambda_1 w_1^k + \lambda_2 w_2^k) \wedge w^m = \lambda_1 (w_1^k \wedge w^m) + \lambda_2 (w_2^k \wedge w^m).
    \]

    \begin{proof}
    This follows directly from the linearity of the wedge product in each argument. By expanding the left-hand side, we obtain
    \[
    (\lambda_1 w_1^k + \lambda_2 w_2^k) \wedge w^m = \lambda_1 w_1^k \wedge w^m + \lambda_2 w_2^k \wedge w^m,
    \]
    which matches the right-hand side.
    \end{proof}
\end{enumerate}

\begin{theorem}\label{thm:differential_one_form}

\end{theorem}

\begin{proof}

are smooth.
\end{proof}

\begin{definition}\label{def:isotropic}
Let $(V, \omega)$ be a symplectic linear space. Let $W$ be a subspace of $V$. We call $W$ to be \textbf{isotropic} if
\[ W \subseteq W^\perp = \{ v \in V \mid \omega(v, w) = 0 \ \forall w \in W \}. \]
\end{definition}

\begin{example}\label{ex:1d_isotropic}
Any 1-dimensional subspace of $V$ is isotropic.
\end{example}

\begin{proof}
Assume that $W = \langle w_0 \rangle$, then
\[ W^\perp := \{ v \in V \mid \omega(v, w_0) = 0 \}. \]
Since $W$ is 1-dimensional, for any $w \in W$, $w = \lambda w_0$ for some $\lambda \in \mathbb{R}$. Then,
\[ \omega(w, w) = \omega(\lambda w_0, \lambda w_0) = \lambda^2 \omega(w_0, w_0) = 0, \]
because $\omega$ is skew-symmetric and $\omega(w_0, w_0) = 0$. Thus, $W \subseteq W^\perp$.
\end{proof}

\begin{remark}\label{rem:properties_of_W_perp}
Properties of $W^\perp$:
\begin{itemize}
\item $W^\perp = \{ v \in V \mid \omega(v, w) = 0 \ \forall w \in W \}$.
\item $W^\perp$ is a subspace of $V$.
\item Observe that $\dim W + \dim W^\perp = \dim V$.
\end{itemize}
\end{remark}

\begin{proof}[Proof of the observation]
Consider the map
\[ I \colon V \to W^*, \]
\[ v \mapsto \omega(v, \cdot) \colon W \to \mathbb{R}. \]
The kernel of $I$ is $W^\perp$. By the rank-nullity theorem,
\[ \dim V = \dim \ker I + \dim \operatorname{Im} I = \dim W^\perp + \dim \operatorname{Im} I. \]
Since $\operatorname{Im} I \cong W^*$ and $\dim W^* = \dim W$, we have
\[ \dim V = \dim W^\perp + \dim W. \]
\end{proof}

Let $H: \mathbb{R}^2 \to \mathbb{R}$ be a smooth function


\begin{definition}\label{def:hamflow}
The Hamiltonian vector field $X_H$ associated with $H$ is defined by
\[
X_H = \frac{\partial H}{\partial p_1} \frac{\partial}{\partial q} - \frac{\partial H}{\partial q} \frac{\partial}{\partial p_1}.
\]
\end{definition}

\begin{theorem}\label{thm:area-preservation}
If $A$ is a region in $\mathbb{R}^2$, then the area of $A$ is preserved under the flow of the Hamiltonian vector field $X_H$.
\end{theorem}

\begin{proof}
The flow of the Hamiltonian vector field $X_H$ is given by the Hamiltonian system of differential equations
\[
\frac{dq}{dt} = \frac{\partial H}{\partial p_1}, \quad \frac{dp_1}{dt} = -\frac{\partial H}{\partial q}.
\]
The divergence of $X_H$ is
\[
\nabla \cdot X_H = \frac{\partial}{\partial q}\left(\frac{\partial H}{\partial p_1}\right) + \frac{\partial}{\partial p_1}\left(-\frac{\partial H}{\partial q}\right) = 0,
\]
since $H$ is smooth. By Liouville's theorem, the flow preserves volume (and hence area in $\mathbb{R}^2$).
\end{proof}
