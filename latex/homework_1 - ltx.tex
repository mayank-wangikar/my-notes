Given $m, n \in \mathbb{Z}$ and $d = \gcd(m, n)$, let $I$ be the ideal generated by $\{m, n\}$.

\begin{theorem}\label{thm:ideal_gcd}
    Show that $I = (d)$.
\end{theorem}

\begin{proof}
    Let $X = \{J \subseteq \mathbb{Z} \mid J \text{ is an ideal and } X \subseteq J\}$. Define
    \[
    (X) = \bigcap_{J \in \mathcal{F}} J,
    \]
    where $\mathcal{F} = \{J \subseteq \mathbb{Z} \mid X \subseteq J\}$.

    If $a \in (X)$, then $a \in J$ for every $J \in \mathcal{F}$.

    Since $J$ is an ideal of $\mathbb{Z}$, it must also be a subgroup of $\mathbb{Z}$. Thus, $\exists n_J \in \mathbb{Z}$ such that $J = n_J \mathbb{Z}$.

    If $x \in J$, then $n_J \mid x$ and $n_J \mid y$ for some $x, y \in X$.

    Since $d = \gcd(x, y)$, we have $d \mid n_J$.

    Therefore, $n_J \mathbb{Z} \subseteq d \mathbb{Z}$, where $n_J$ is the integer corresponding to $J$.

    Hence, $(X) \subseteq d \mathbb{Z}$.

    \textbf{Claim:} $d \mathbb{Z}$ is the smallest ideal containing $X$.

    Consider $n \mathbb{Z} \subseteq \mathbb{Z}$ such that $X \subseteq n \mathbb{Z}$ and $n \mathbb{Z} \subseteq d \mathbb{Z}$.

    Let $nk_1 = x$ and $nk_2 = y$ for some $k_1, k_2 \in \mathbb{Z}$.

    By Bézout's identity, there exist $r_1, r_2 \in \mathbb{Z}$ such that
    \[
    x r_1 + y r_2 = d.
    \]
    Substituting $x = nk_1$ and $y = nk_2$, we get
    \[
    n(k_1 r_1 + k_2 r_2) = d.
    \]
    Thus, $n \mid d$.

    Hence, $d \mathbb{Z} \subseteq n \mathbb{Z}$, which implies $n \mathbb{Z} = d \mathbb{Z}$.

    Therefore, $I = (d)$.
\end{proof}

\[ C_c(\mathbb{R}) := \{ f : \mathbb{R} \to \mathbb{C} \mid f \text{ is continuous and has compact support} \}, \]

\[ f(x) = 0 \text{ whenever } x \notin [a, b], \]
and $(+, \times)$ are the same as usual addition and multiplication.
\end{definition}

\begin{remark}\label{rem:ring}
Observe that $C_c(\mathbb{R})$ is a ring, and it is non-empty.

\begin{theorem}\label{thm:nonunit}
Show that $C_c(\mathbb{R})$ is not unital.
\end{theorem}

\begin{proof}
Assume there is $g \in C_c(\mathbb{R})$ such that $f \times g = f$ for any $f \in C_c(\mathbb{R})$.

If $g(x_0) \neq 0$ and $x_0 \in [a, b]$, choose $f \in C_c(\mathbb{R})$ such that
\[ f(x) \neq 0 \text{ whenever } x \in [a', b'] \]

\[ g(x_0) = 0 \quad \text{and} \quad f(x_0) \neq 0 \implies f \times g(x_0) \neq 0. \]

This is a contradiction. Hence, no unit exists in $C_c(\mathbb{R})$.
\end{proof}
