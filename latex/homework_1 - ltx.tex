Given $m, n \in \mathbb{Z}$ and $d = \gcd(m, n)$, let
\[
I = \text{Ideal generated by } \{m, n\}.
\]

\begin{theorem}\label{thm:ideal_gcd}
Show that $I = (d)$.
\end{theorem}

\begin{proof}
Let $X = \{J \subseteq \mathbb{Z} \mid J \text{ is an ideal and } X \subseteq J\}$.
Define
\[
C(X) = \bigcap_{J \in \mathcal{F}} J,
\]
where $\mathcal{F} = \{J \subseteq \mathbb{Z} \mid X \subseteq J\}$.

If $a \in C(X)$, then $a \in J$ for every $J \in \mathcal{F}$.

Since $J$ is an ideal of $\mathbb{Z}$, it must also be a subgroup of $\mathbb{Z}$. Thus, $\exists n \in \mathbb{Z}$ such that $J = n\mathbb{Z}$.

If $x \in n\mathbb{Z}$, then $n \mid x$ and $n \mid y$ for some $y \in \mathbb{Z}$.

Since $d = \gcd(x, y)$, we have $d \mid n$.

Thus, $n\mathbb{Z} \subseteq d\mathbb{Z}$, where $n$ corresponds to $J$.

Therefore, $C(X) \subseteq d\mathbb{Z}$.

\begin{claim}
$d\mathbb{Z}$ is the smallest ideal containing $X$.
\end{claim}

Consider $n\mathbb{Z} \subseteq \mathbb{Z}$ such that $X \subseteq n\mathbb{Z}$ and $n\mathbb{Z} \subseteq d\mathbb{Z}$.

Let $nk_1 = x$ and $nk_2 = y$ for some $k_1, k_2 \in \mathbb{Z}$.

By Bézout's identity, we have
\[
nk_1 r_1 + nk_2 r_2 = x r_1 + y r_2 = d,
\]
since $d = \gcd(x, y)$ and $k_1, k_2, r_1, r_2 \in \mathbb{Z}$.

Thus, $n \mid d$, which implies $d\mathbb{Z} \subseteq n\mathbb{Z}$.

Hence, $n\mathbb{Z} = d\mathbb{Z}$.

Therefore, $I = (d)$.
\end{proof}

\begin{definition}\label{def:cc}
Given the set
\[ C_c(\mathbb{R}) := \{ f : \mathbb{R} \to \mathbb{C} \mid f \text{ is continuous and has compact support} \}, \]
where compact support means $\exists a, b \in \mathbb{R}$ such that
\[ f(x) = 0 \quad \text{whenever} \quad x \notin [a, b]. \]
The operations $+$ and $\times$ are the usual addition and multiplication of functions.
\end{definition}

\begin{remark}\label{rem:ring}
Observe that $C_c(\mathbb{R})$ is a ring and is non-empty.
\end{remark}

\begin{theorem}\label{thm:nonunit}
Show that $C_c(\mathbb{R})$ is \textbf{not} unital.
\end{theorem}

\begin{proof}
Assume there is a $g \in C_c(\mathbb{R})$ such that
\[ f \times g = f \quad \text{for any} \quad f \in C_c(\mathbb{R}). \]

If $g(x_0) \neq 0$ for some $x_0 \in [a, b]$, choose $f \in C_c(\mathbb{R})$ such that
\[ f(x) \neq 0 \quad \text{for some} \quad x \in [a', b'] \]
where $[a, b] \subset [a', b']$. Then, observe at $x_0 = \frac{a' + a}{2}$,
\[ g(x_0) = 0 \quad \text{and} \quad f(x_0) \neq 0 \implies f \times g(x_0) \neq 0. \]

This leads to a contradiction. Hence, no unit exists in $C_c(\mathbb{R})$.
\end{proof}

Given: $D \in \mathbb{Q}$ be a rational number and is NOT a perfect square.

\[ R := \{ a + b \sqrt{D} \mid a, b \in \mathbb{Q} \} \]

(a)
\begin{proof}
To show $R$ is a subring of $\mathbb{C}$, we need to verify the following:

1. $R \neq \emptyset$ as $0 \in R$.

Consider $x, y \in R$, where $x = a_1 + b_1 \sqrt{D}$ and $y = a_2 + b_2 \sqrt{D}$ with $a_1, b_1, a_2, b_2 \in \mathbb{Q}$.

For closure under subtraction:
\[ x - y = (a_1 - a_2) + (b_1 - b_2) \sqrt{D} \in R \]
since $a_1 - a_2 \in \mathbb{Q}$ and $b_1 - b_2 \in \mathbb{Q}$.

Thus, $(R, +)$ is a subgroup of $(\mathbb{C}, +)$.

For closure under multiplication:
\[ xy = (a_1 + b_1 \sqrt{D})(a_2 + b_2 \sqrt{D}) = a_1 a_2 + b_1 b_2 D + (a_2 b_1 + a_1 b_2) \sqrt{D} \]
and $a_1 a_2 + b_1 b_2 D, a_2 b_1 + a_1 b_2 \in \mathbb{Q}$.

Hence, $xy \in R$ and $(R, +, \cdot)$ is a subring of $(\mathbb{C}, +, \cdot)$.
\end{proof}

(b)
\begin{theorem}\label{thm:nonzero_element}
If $(a + b \sqrt{D}) \neq 0$, then show that $a^2 - b^2 D \neq 0$.
\end{theorem}

\begin{proof}
If $(a + b \sqrt{D}) \neq 0$, then $a \neq 0$ or $b \neq 0$. Hence, $a - b \sqrt{D} \neq 0$. Now,
\[ (a + b \sqrt{D})(a - b \sqrt{D}) \neq 0 \]
implies
\[ a^2 - b^2 D \neq 0. \]
\end{proof}

(c)
\begin{theorem}\label{thm:field_property}
Prove: $R$ is a field.
\end{theorem}

\begin{proof}
Observe $R$ is a commutative, unital ring. Moreover, $R$ is a division ring as for $a + b \sqrt{D} \neq 0$, then
\[ \frac{a - b \sqrt{D}}{a^2 - b^2 D} \]
is the inverse with respect to multiplication.

Now, $R$ is a commutative division ring, which is a field.
\end{proof}

\begin{definition}\label{def:zcr}
Let $R$ be a unital ring. The \textbf{center} of $R$, denoted $Z(R)$, is defined as
\[ Z(R) = \{ a \in R \mid ab = ba \text{ for all } b \in R \}. \]
\end{definition}

\begin{proof}
To show that $Z(R)$ is a subring of $R$, we proceed as follows:

First, observe that $Z(R) \neq \emptyset$ since $1 \in Z(R)$.

Next, consider $a, a' \in Z(R)$. For any $b \in R$,
\[
(a - a')b = ab - a'b = ba - ba' = b(a - a'),
\]
which implies that $(Z(R), +)$ is a subgroup of $R$.

For any $b \in R$, observe that
\[
(aa')b = a(a'b) = a(ba') = (ab)a' = (ba)a' = b(aa'),
\]
which implies that $aa' \in Z(R)$.

Hence, $Z(R)$ is a subring of $R$.
\end{proof}
