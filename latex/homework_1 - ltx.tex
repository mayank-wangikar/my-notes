Given $m, n \in \mathbb{Z}$ and $d = \gcd(m, n)$,

\begin{definition}\label{def:ideal_generated}
Let $I$ be the ideal generated by $\{m, n\}$.
\end{definition}

\begin{theorem}\label{thm:ideal_gcd}
Show that $I = (d)$.
\end{theorem}

\begin{proof}
Consider $C(X) = \bigcap_{J \in \mathcal{F}} J$ where
\[
\mathcal{F} = \{ J \triangleleft \mathbb{Z} \mid X \subseteq J \}.
\]
If $a \in C(X)$, then $a \in J$ for every $J \in \mathcal{F}$.

Since $J$ is an ideal of $\mathbb{Z}$, it must also be a subgroup of $\mathbb{Z}$, so there exists $n_J \in \mathbb{Z}$ such that $J = n_J \mathbb{Z}$.

If $X \subseteq n_J \mathbb{Z}$, then $n_J \mid x$ and $n_J \mid y$ for $x, y \in X$.

Since $d = \gcd(x, y)$, we have $d \mid n_J$.

Thus, $n_J \mathbb{Z} \supseteq d \mathbb{Z}$ for every $J \in \mathcal{F}$, which implies $C(X) \supseteq d \mathbb{Z}$.

\begin{claim}
$d \mathbb{Z}$ is the smallest ideal containing $X$.
\end{claim}

Consider $n \mathbb{Z} \triangleleft \mathbb{Z}$ such that $X \subseteq n \mathbb{Z}$ and $n \mathbb{Z} \subseteq d \mathbb{Z}$.

Let $x = n k_1$ and $y = n k_2$ for some $k_1, k_2 \in \mathbb{Z}$.

By BĂ©zout's identity, there exist $r_1, r_2 \in \mathbb{Z}$ such that
\[
x r_1 + y r_2 = d.
\]
Substituting $x$ and $y$ gives
\[
n(k_1 r_1 + k_2 r_2) = d.
\]
Thus, $n \mid d$, which implies $d \mathbb{Z} \subseteq n \mathbb{Z}$.

Hence, $n \mathbb{Z} = d \mathbb{Z}$.
\end{proof}

$\square$

\begin{definition}\label{def:ccir}
Let $C_c(\mathbb{R})$ be the set of continuous functions $f: \mathbb{R} \to \mathbb{C}$ with compact support, where compact support means that there exist $a, b \in \mathbb{R}$ such that
\[ f(x) = 0 \quad \text{whenever} \quad x \notin [a, b]. \]
The operations $+$ and $\times$ are defined as usual addition and multiplication of functions.
\end{definition}

\begin{remark}\label{rem:ccirring}
Observe that $C_c(\mathbb{R})$ is a ring and is non-empty.
\end{remark}

\begin{theorem}\label{thm:nonunit}
Show that $C_c(\mathbb{R})$ is not unital.
\end{theorem}

\begin{proof}
Assume there is a $g \in C_c(\mathbb{R})$ such that $f \times g = f$ for any $f \in C_c(\mathbb{R})$.

If $g(x_0) \neq 0$ for some $x_0 \in [a, b]$, choose $f \in C_c(\mathbb{R})$ such that
\[ f(x) \neq 0 \quad \text{for some} \quad x \in [a', b'] \]
where $[a, b] \subset [a', b']$. Then observe at $x_0 = \frac{a' + a}{2}$,
\[ g(x_0) = 0 \quad \text{and} \quad f(x_0) \neq 0 \implies f \times g(x_0) \neq 0. \]

This leads to a contradiction. Hence, no unit exists in $C_c(\mathbb{R})$.
\end{proof}

\begin{definition}\label{def:zcr}
Let $R$ be a unital ring. The \emph{center} of $R$, denoted $Z(R)$, is defined as
\[ Z(R) = \{ a \in R \mid ab = ba \text{ for all } b \in R \}. \]
\end{definition}

\begin{proof}
To show that $Z(R)$ is a subring of $R$, we proceed as follows:

First, observe that $Z(R) \neq \emptyset$ since $1 \in Z(R)$.

Now, consider $a, a' \in Z(R)$. For any $b \in R$,
\[
(a - a')b = ab - a'b = ba - ba' = b(a - a'),
\]
which implies $(Z(R), +)$ is a subgroup of $(R, +)$.

For any $b \in R$, observe
\[
(aa')b = a(a'b) = a(ba') = (ab)a' = (ba)a' = b(aa'),
\]
which implies $aa' \in Z(R)$.

Hence, $Z(R)$ is a subring of $R$.
\end{proof}
