\begin{definition}\label{def:random_matrix}
Let \(X\) be a random matrix with entries \(X_{ij}\) such that
\[
X = \begin{pmatrix}
* & \cdots & * \\
\vdots & \ddots & \vdots \\
* & \cdots & *
\end{pmatrix},
\]
where \(X_1, \dots, X_n\) are random vectors in \(\mathbb{R}^d\) with components \(X_i = (X_{i1}, \dots, X_{id})\).
\end{definition}

\begin{definition}\label{def:variance}
The sample variance for the \(i\)-th component is given by
\[
\frac{1}{n} \sum_{i=1}^n X_{i,i}^2.
\]
\end{definition}

\begin{definition}\label{def:covariance}
The sample covariance between components \(X_{i1}\) and \(X_{i2}\) is given by
\[
\frac{1}{n} \sum_{i=1}^n X_{i1} X_{i2}.
\]
\end{definition}

Let \(X\) be the data matrix with entries \(X_{ij}\):
\[
X = \begin{pmatrix}
X_{11} & \cdots & X_{1d} \\
\vdots & \ddots & \vdots \\
X_{n1} & \cdots & X_{nd}
\end{pmatrix}.
\]
The sample covariance matrix is defined as
\[
\frac{XX^\top}{n}.
\]

\begin{definition}\label{def:eigenvalue}
The eigenvalues \(\lambda_i\) of a matrix \(A\) satisfy
\[
Av = \lambda v \quad \text{for} \quad v \neq 0.
\]
The trace of \(A\) is given by
\[
\text{tr}(A) = \sum_i A_{ii} = \sum_i \lambda_i.
\]
\end{definition}

\begin{remark}\label{rem:spectral_distribution}
The set of eigenvalues of a random matrix is called the \emph{empirical spectral distribution}. It is defined as
\[
\frac{1}{n} \sum_{i=1}^n \delta_{\lambda_i},
\]
where \(\delta_{\lambda_i}\) is the Dirac delta function at \(\lambda_i\).
\end{remark}

\begin{definition}\label{def:random_matrix}
Let \( A \) be an \( n \times n \) matrix with entries \( A_{ij} \) such that each \( A_{ij} \) is independently chosen to be \( 1 \) or \( -1 \) with probability \( \frac{1}{2} \). The matrix \( A \) is symmetric, i.e., \( A_{ji} = A_{ij} \).
\end{definition}

\begin{remark}\label{rem:trace_properties}
For such a matrix \( A \), the following properties hold for its eigenvalues \( \lambda_i \):

\[
\text{tr}(A) = \sum_{i=1}^n A_{ii} = \sum_{i=1}^n \lambda_i,
\]

\[
\text{tr}(A^2) = \sum_{i=1}^n A_{ii}^2 = \sum_{i=1}^n \lambda_i^2,
\]

\[
\text{tr}(A^k) = \sum_{i=1}^n A_{ii}^k = \sum_{i=1}^n \lambda_i^k.
\]
\end{remark}
