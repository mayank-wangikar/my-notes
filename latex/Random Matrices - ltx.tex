\begin{definition}\label{def:random_matrix}
Consider a matrix of random entries:
\[
\begin{pmatrix}
* & \cdots & * \\
\vdots & \ddots & \vdots \\
* & \cdots & *
\end{pmatrix}
\]
These entries are random variables. Let \(X_1, \ldots, X_n\) be vectors in \(\mathbb{R}^d\), where each \(X_i = (X_{i1}, \ldots, X_{id})\) are random variables.
\end{definition}

\begin{definition}\label{def:variance_covariance}
The variance of a random variable \(X_{ij}\) is defined as:
\[
\frac{1}{n} \sum_{i=1}^n X_{ii}^2 \quad \longrightarrow \quad \text{Variance}
\]
The covariance between two random variables \(X_{i1}\) and \(X_{i2}\) is defined as:
\[
\frac{1}{n} \sum_{i=1}^n X_{i1} X_{i2} \quad \longrightarrow \quad \text{Covariance}
\]
\end{definition}

Let \(X\) be the data matrix:
\[
X = \begin{pmatrix}
X_{11} & \cdots & X_{1d} \\
\vdots & \ddots & \vdots \\
X_{n1} & \cdots & X_{nd}
\end{pmatrix}
\]
The sample covariance matrix is given by:
\[
\frac{XX^\top}{n}
\]

\begin{definition}\label{def:eigenvalues}
The eigenvalues \(\lambda\) of a matrix \(A\) are defined by the equation:
\[
Av = \lambda v \quad \text{for} \quad v \neq 0
\]
The trace of a matrix \(A\) is the sum of its eigenvalues:
\[
\text{tr}(A) = \sum_{i=1}^n A_{ii} = \sum_{i=1}^n \lambda_i
\]
\end{definition}

\begin{remark}\label{rem:spectral_distribution}
The set of eigenvalues of a random matrix is called the empirical spectral distribution:
\[
\frac{1}{n} \sum_{i=1}^n \delta_{\lambda_i}
\]
\end{remark}
