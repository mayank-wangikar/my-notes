\begin{definition}\label{def:random_matrix}
Consider a random matrix with entries
\[
\begin{pmatrix}
* & \cdots & * \\
\vdots & \ddots & \vdots \\
* & \cdots & *
\end{pmatrix}
\]
where these entries are random. Let $X_1, \ldots, X_n$ be vectors in $\mathbb{R}^d$, where each $X_i = (X_{i1}, \ldots, X_{id})$ consists of random variables.
\end{definition}

The variance of the $i$-th component is given by
\[
\frac{1}{n} \sum_{i=1}^n X_{ii}^2 \quad \longrightarrow \quad \text{Variance}.
\]

The covariance between two components is given by
\[
\frac{1}{n} \sum_{i=1}^n X_{i1} X_{i2} \quad \longrightarrow \quad \text{Covariance}.
\]

Let $X$ be the data matrix
\[
X = \begin{pmatrix}
X_{11} & \cdots & X_{1d} \\
\vdots & \ddots & \vdots \\
X_{n1} & \cdots & X_{nd}
\end{pmatrix}.
\]
The sample covariance matrix is defined as
\[
\frac{XX^\top}{n}.
\]

\begin{definition}\label{def:eigenvalues}
The eigenvalues $\lambda$ of a matrix $A$ are defined by the equation
\[
Av = \lambda v \quad \text{for} \quad v \neq 0.
\]
The trace of a matrix $A$ is given by
\[
\text{tr}(A) = \sum_{i} A_{ii} = \sum_{i} \lambda_i,
\]
where $\lambda_i$ are the eigenvalues of $A$.
\end{definition}

\begin{remark}\label{rem:spectral_distribution}
The set of eigenvalues of a random matrix is referred to as the empirical spectral distribution, denoted by
\[
\frac{1}{n} \sum_{i=1}^n \delta_{\lambda_i}.
\]
\end{remark}

Consider a matrix \( A \) with entries \( A_{ij} \) such that:

\[
A = \begin{pmatrix}
A_{11} & A_{1j} & \cdots \\
A_{j1} & A_{jj} & \cdots \\
\vdots & \vdots & \ddots \\
A_{m1} & A_{mj} & \cdots & A_{mn}
\end{pmatrix}
\]

where the entries \( A_{ij} \) are independent random variables with probability \( \frac{1}{2} \) for \( 1 \) or \( -1 \), and \( A_{ji} = A_{ij} \).

\[
\begin{definition}\label{def:trace_properties}
    For such a matrix \( A \), the following properties hold for its trace and powers:
    \[
    \tr(A) = \sum_i A_{ii} = \sum_i \lambda_i
    \]
    \[
    \tr(A^2) = \sum_i A_{ii}^2 = \sum_i \lambda_i^2
    \]
    \[
    \tr(A^k) = \sum_i A_{ii}^k = \sum_i \lambda_i^k
    \]
\end{definition}
