\begin{itemize}
    \item \textbf{Vector subspaces}

    \item In $\mathbb{Z}$-modules say $M$
    \begin{definition}\label{def:submodule_z}
        $N$ is a submodule iff $(N, +) \leq (CM, +)$.
    \end{definition}

    \item In $R$ is a $R$-module
    \begin{definition}\label{def:submodule_ring}
        $N$ is a submodule iff $N$ is an ideal.
    \end{definition}

    \item If $M$ is a $F[x]$-module $\cong (V, T)$
    \begin{definition}\label{def:submodule_fx}
        $W$ is a submodule iff $T(CW) \subseteq W$, i.e., $W$ is invariant under $T$.
    \end{definition}
\end{itemize}

\begin{definition}\label{def:finitely_generated_module}
A module $(V, \mathbb{F})$ is said to be finitely generated if its dimension $\dim V$ is finite, i.e., $\dim V < +\infty$.
\end{definition}

\begin{definition}\label{def:cyclic_module}
A module $(V, \mathbb{F})$ is said to be cyclic if its dimension $\dim V$ is either $0$ or $1$, i.e., $\dim V = \{0, 1\}$.
\end{definition}

\begin{definition}\label{def:finitely_generated_group}
An abelian group $G$, treated as a $\mathbb{Z}$-module, is finitely generated as a group if it is finitely generated as a $\mathbb{Z}$-module.
\end{definition}

\begin{definition}\label{def:cyclic_group}
An abelian group $G$, treated as a $\mathbb{Z}$-module, is cyclic as a group if it is cyclic as a $\mathbb{Z}$-module.
\end{definition}

\begin{definition}\label{def:finitely_generated_ideal}
Let $R$ be a commutative ring and $M$ an $R$-module such that $M \triangleleft R$. The module $M$ is finitely generated as an ideal if it is finitely generated as an $R$-module.
\end{definition}

\begin{definition}\label{def:principal_ideal}
Let $R$ be a commutative ring and $M$ an $R$-module such that $M \triangleleft R$. The module $M$ is a principal ideal if it is cyclic as an $R$-module.
\end{definition}

\begin{itemize}
    \item \textbf{Module homomorphisms}

    \begin{itemize}
        \item Abelian groups $\leftrightarrow$ $\mathbb{Z}$-modules, then group homomorphisms $\leftrightarrow$ module homomorphisms.
    \end{itemize}

    \begin{itemize}
        \item For vector fields, linear transformations $\leftrightarrow$ module homomorphisms.
    \end{itemize}

    \begin{itemize}
        \item For a ring $R$, $\Theta: M \to N$ where $M = N = R$ and
        \[
        \Theta(rs) = r\Theta(s)
        \]
        is \textbf{not} a ring homomorphism but is a module homomorphism.
    \end{itemize}

    \item Isomorphism theorems hold for modules.
\end{itemize}
