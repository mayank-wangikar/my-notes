    \begin{enumerate}
        \item Vector space $\mathbb{C}$ (Ring over module is a field).
        \item $(V, T)$ where $V$ is a vector space and $T$ is a linear transform. Ring is constructed to be $F[x]$ where
        \[
        f(x) \cdot v := f(T)(v).

\begin{itemize}
    \item \textbf{Vector subspaces}

    \item In $\mathbb{Z}$-modules say $M$
    \begin{definition}\label{def:submodule_z}
        $N$ is a submodule iff $(N, +) \leq (CM, +)$.
    \end{definition}

    \item In $R$ is a $R$-module
    \begin{definition}\label{def:submodule_ring}
        $N$ is a submodule iff $N$ is an ideal.
    \end{definition}

    \item If $M$ is a $F[x]$-module $\cong (V, T)$
    \begin{definition}\label{def:submodule_fx}
        $W$ is a submodule iff $T(CW) \subseteq W$, i.e., $W$ is invariant under $T$.
    \end{definition}
\end{itemize}

\begin{definition}\label{def:finitely-generated-module}
A module $(V, \mathbb{F})$ is \textit{finitely generated} if $\dim V < +\infty$.
\end{definition}

\begin{definition}\label{def:cyclic-module}
A module $(V, \mathbb{F})$ is \textit{cyclic} if $\dim V = \{0, 1\}$.
\end{definition}

\begin{definition}\label{def:finitely-generated-abelian-group}
An abelian group $G$, treated as a $\mathbb{Z}$-module, is \textit{finitely generated} as a group if it is finitely generated as a group.
\end{definition}

\begin{definition}\label{def:cyclic-abelian-group}
An abelian group $G$, treated as a $\mathbb{Z}$-module, is \textit{cyclic} as a group if it is cyclic as a group.
\end{definition}

\begin{definition}\label{def:finitely-generated-ideal}
Let $R$ be a commutative ring and $M$ an $R$-module. $M$ is \textit{finitely generated} as an ideal if it is finitely generated as an ideal.
\end{definition}

\begin{definition}\label{def:principal-ideal}
Let $R$ be a commutative ring and $M$ an $R$-module. $M$ is a \textit{principal ideal} if it is generated by a single element.
\end{definition}

\begin{itemize}
    \item \textbf{Module homomorphisms}

    \begin{itemize}
        \item Abelian groups $\leftrightarrow$ $\mathbb{Z}$-modules, then group homomorphisms $\leftrightarrow$ module homomorphisms.
    \end{itemize}

    \begin{itemize}
        \item For vector fields, linear transformations $\leftrightarrow$ module homomorphisms.
    \end{itemize}

    \begin{itemize}
        \item For a ring $R$, a map $\Theta: M \to N$ where $M = N = R$ and
        \[
        \Theta(rs) = r\Theta(s)
        \]
        is \textbf{not} a ring homomorphism but is a module homomorphism.
    \end{itemize}

    \item Isomorphism theorems hold for modules.
\end{itemize}
