\subsection*{Submodules}

\begin{itemize}
    \item \textbf{Vector subspaces}

    \item In $\mathbb{Z}$-modules say $M$
    \begin{definition}\label{def:submodule_z}
        $N$ is a submodule iff $(N, +) \leq (CM, +)$.
    \end{definition}

    \item In $R$ is a $R$-module
    \begin{definition}\label{def:submodule_ring}
        $N$ is a submodule iff $N$ is an ideal.
    \end{definition}

    \item If $M$ is a $F[x]$-module $\cong (V, T)$
    \begin{definition}\label{def:submodule_fx}
        $W$ is a submodule iff $T(CW) \subseteq W$, i.e., $W$ is invariant under $T$.
    \end{definition}
\end{itemize}

\begin{center}
\textbf{Finitely generated and cyclic modules}
\end{center}

\begin{table}[h]
\begin{tabular}{|p{4cm}|p{4cm}|p{4cm}|}
\hline
\textbf{Module} & \textbf{f.g.} & \textbf{cyclic} \\
\hline
$(V, \mathcal{F})$ & $\dim V < +\infty$ & $\dim V = \{0, 1\}$ \\
\hline
$G$ is abelian group treated like $\mathbb{Z}$-module & f.g. as group & cyclic as group \\
\hline
$R$ comm ring and $M \triangleleft R$, $M$ is $R$-module & f.g. as an ideal & principal ideal \\
\hline
\end{tabular}
\end{table}

\begin{definition}\label{def:finitely_generated_module}
A module $(V, \mathcal{F})$ is said to be \textit{finitely generated} if $\dim V < +\infty$.
\end{definition}

\begin{definition}\label{def:cyclic_module}
A module $(V, \mathcal{F})$ is said to be \textit{cyclic} if $\dim V = \{0, 1\}$.
\end{definition}

\begin{definition}\label{def:finitely_generated_abelian_group}
An abelian group $G$ treated as a $\mathbb{Z}$-module is \textit{finitely generated} as a group if it is finitely generated as a group.
\end{definition}

\begin{definition}\label{def:cyclic_abelian_group}
An abelian group $G$ treated as a $\mathbb{Z}$-module is \textit{cyclic} as a group if it is cyclic as a group.
\end{definition}

\begin{definition}\label{def:finitely_generated_ideal}
Let $R$ be a commutative ring and $M$ an ideal of $R$ such that $M$ is an $R$-module. The ideal $M$ is \textit{finitely generated} as an ideal.
\end{definition}

\begin{definition}\label{def:principal_ideal}
Let $R$ be a commutative ring and $M$ an ideal of $R$ such that $M$ is an $R$-module. The ideal $M$ is \textit{principal} if it is generated by a single element.
\end{definition}

\begin{itemize}
    \item \textbf{Module homomorphisms}

    \begin{definition}\label{def:module_homomorphism}
        \begin{enumerate}
            \item Abelian groups $\leftrightarrow$ $\mathbb{Z}$-modules, then group homomorphism $\leftrightarrow$ module homomorphism.
            \item For vector fields, linear transformation $\leftrightarrow$ module homomorphism.
            \item For ring $R$, $\Theta: M \to N$ where $M = N = R$ and $\Theta(rs) = r\Theta(s)$ is \textbf{not} a ring homomorphism but is a module homomorphism.
        \end{enumerate}
    \end{definition}

    \begin{remark}\label{rem:isomorphism_theorems}
        Isomorphism theorems hold for modules.
    \end{remark}
\end{itemize}
