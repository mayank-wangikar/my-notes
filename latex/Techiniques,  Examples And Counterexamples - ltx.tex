\begin{itemize}
    \item \textbf{Submodules}

    \begin{definition}\label{def:submodule_Z}
        In $\mathbb{Z}$-modules, say $M$, $N$ is a submodule if $(N, +) \leq (CM, +)$.
    \end{definition}

    \begin{definition}\label{def:submodule_R}
        In $R$ is a $R$-module, $N$ is a submodule if $N$ is an ideal.
    \end{definition}

    \begin{definition}\label{def:submodule_Fx}
        If $M$ is a $F[x]$-module $\cong (V, T)$, $W$ is a submodule if $T(CW) \subseteq W$, i.e., $W$ is invariant under $T$.
    \end{definition}
\end{itemize}

\begin{definition}\label{def:finitely-generated-module}
Let $(V, \mathbb{F})$ be a vector space over a field $\mathbb{F}$. The vector space $V$ is said to be \textit{finitely generated} if there exists a finite set of vectors $\{v_1, v_2, \ldots, v_n\}$ such that every vector in $V$ can be written as a linear combination of these vectors. In other words, $\dim V < \infty$.

A vector space $V$ is said to be \textit{cyclic} if it is generated by a single vector, i.e., $\dim V = \{0, 1\}$.
\end{definition}

\begin{definition}\label{def:finitely-generated-abelian-group}
Let $G$ be an abelian group. When $G$ is treated as a $\mathbb{Z}$-module, it is said to be \textit{finitely generated} if there exists a finite set of elements $\{g_1, g_2, \ldots, g_n\}$ such that every element of $G$ can be written as a linear combination of these elements with integer coefficients. In other words, $G$ is finitely generated as a group.

A group $G$ is said to be \textit{cyclic} if it is generated by a single element, i.e., $G$ is cyclic as a group.
\end{definition}

\begin{definition}\label{def:finitely-generated-ideal}
Let $R$ be a commutative ring and $M$ an ideal of $R$. The ideal $M$ is said to be \textit{finitely generated} if there exists a finite set of elements $\{m_1, m_2, \ldots, m_n\}$ in $M$ such that every element of $M$ can be written as a linear combination of these elements with coefficients in $R$. In other words, $M$ is finitely generated as an ideal.

An ideal $M$ is said to be \textit{principal} if it is generated by a single element, i.e., $M$ is a principal ideal.
\end{definition}

\begin{itemize}
    \item \textbf{Module homomorphisms}

    \begin{itemize}
        \item Abelian groups $\leftrightarrow$ $\mathbb{Z}$-modules, then group homomorphisms $\leftrightarrow$ module homomorphisms.
    \end{itemize}

    \begin{itemize}
        \item For vector fields, linear transformations $\leftrightarrow$ module homomorphisms.
    \end{itemize}

    \begin{itemize}
        \item For a ring $R$, $\Theta: M \to N$ where $M = N = R$ and
        \[
        \Theta(rs) = r\Theta(s)
        \]
        is \textbf{not} a ring homomorphism but is a module homomorphism.
    \end{itemize}

    \item Isomorphism theorems hold for modules.
\end{itemize}
