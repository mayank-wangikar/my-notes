\item \textbf{Measurable function.}

  \begin{itemize}
  \item If $(X, \mathcal{M})$ and $(Y, \mathcal{N})$ are measurable spaces a mapping $f: X \rightarrow Y$ is called $(\mathcal{M}, \mathcal{N})$-measurable or \textbf{measurable} if $f^{-1}(E) \in \mathcal{M}$ for all $E \in \mathcal{N}$.

  \begin{remark}
    Composition of measurable mappings are measurable.
  \end{remark}

  \begin{proposition}
    \label{prop:measurable_generated}
    If $\mathcal{N}$ is generated by $\mathcal{E}$, then $f: X \rightarrow Y$ is $(\mathcal{M}, \mathcal{N})$-measurable iff $f^{-1}(E) \in \mathcal{M}$ for all $E \in \mathcal{E}$.
  \end{proposition}

  \begin{proof}
    Only if condition is trivial. For if condition, observe $\mathcal{E} \subseteq \{E \subseteq Y \mid f^{-1}(E) \in \mathcal{M}\}$ is a $\sigma$-algebra, that contains $\mathcal{E}$ by hypothesis; it therefore contains $\mathcal{N}$.
  \end{proof}
