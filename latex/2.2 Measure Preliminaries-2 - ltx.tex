\begin{definition}\label{def:measurable-function}
Let $(X, \mathcal{M})$ and $(Y, \mathcal{N})$ be measurable spaces. A mapping $f \colon X \to Y$ is called $(\mathcal{M}, \mathcal{N})$-measurable or \underline{measurable} if $f^{-1}(E) \in \mathcal{M}$ for all $E \in \mathcal{N}$.
\end{definition}

\begin{remark}\label{rem:composition-measurable}
The composition of measurable mappings is measurable.
\end{remark}

\begin{proposition}\label{prop:measurable-generated}
If $\mathcal{N}$ is generated by $\mathcal{E}$, then $f \colon X \to Y$ is $(\mathcal{M}, \mathcal{N})$-measurable if and only if $f^{-1}(E) \in \mathcal{M}$ for all $E \in \mathcal{E}$.
\end{proposition}

\begin{proof}
The only if condition is trivial. For the if condition, observe that
\[
\{ E \subseteq Y \mid f^{-1}(E) \in \mathcal{M} \}
\]
is a $\sigma$-algebra that contains $\mathcal{E}$ by hypothesis; it therefore contains $\mathcal{N}$.
\end{proof}

\begin{definition}\label{def:measurable_function}
Let $(X, \mathcal{M})$ be a measurable space. A real or complex-valued function $f$ on $X$ is called \textbf{measurable} if $f$ is $(\mathcal{M}, \mathcal{B}_{\mathbb{R}})$-measurable (for real-valued $f$) or $(\mathcal{M}, \mathcal{B}_{\mathbb{C}})$-measurable (for complex-valued $f$).
\end{definition}
